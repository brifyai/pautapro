  // Factores de puntuación (versión mejorada)
  scoringFactors: {

    retention: { weight: 0.12, maxScore: 12 }, // Retención y lealtad
    diversity: { weight: 0.08, maxScore: 8 }, // Diversidad de medios/campañas
    growth: { weight: 0.05, maxScore: 5 } // Crecimiento potencial
  },
 
  // Niveles de cliente basados en puntaje (mejorado)
  clientLevels: {

  // Calcular scoring completo para un cliente
  async calculateClientScore(clientId) {
    try {
      const scores = await Promise.all([
        this.calculateRecencyScore(clientId),
        this.calculateFrequencyScore(clientId),
        this.calculateMonetaryScore(clientId),
        this.calculateCampaignSuccessScore(clientId),
        this.calculateRetentionScore(clientId)
      ]);
 
      const [recencyScore, frequencyScore, monetaryScore, campaignScore, retentionScore] = scores;
 
      const totalScore = Math.round(
        recencyScore * this.scoringFactors.recency.weight +
        frequencyScore * this.scoringFactors.frequency.weight +
        monetaryScore * this.scoringFactors.monetary.weight +
        campaignScore * this.scoringFactors.campaignSuccess.weight +
        retentionScore * this.scoringFactors.retention.weight
      );
 
      const level = this.getClientLevel(totalScore);
 
      const scoringData = {
        clientId,
        totalScore,
        level: level.label,
        levelColor: level.color,
        breakdown: {
          recency: recencyScore,
          frequency: frequencyScore,
          monetary: monetaryScore,
          campaignSuccess: campaignScore,
          retention: retentionScore
        },
        benefits: level.benefits,
        calculatedAt: new Date().toISOString(),
        recommendations: this.generateRecommendations(totalScore, scores)
      };
 
      // Guardar scoring en base de datos
      await this.saveClientScoring(scoringData);
 
      return scoringData;
    } catch (error) {
      console.error('Error calculando scoring de cliente:', error);
      throw error;
    }
  },
 
  // Calcular puntaje de recencia (última orden)
  async calculateRecencyScore(clientId) {
    try {
      const { data: lastOrder, error } = await supabase
        .from('ordenesdepublicidad')
        .select('created_at')
        .eq('campania.clientes.id_cliente', clientId)
        .order('created_at', { ascending: false })
        .limit(1)
        .single();
 
      if (error && error.code !== 'PGRST116') {
        console.warn('Error calculando recency score, usando valor por defecto:', error);
        return 15; // Valor medio por defecto
      }
 
      if (!lastOrder) return 0;
 
      const lastOrderDate = new Date(lastOrder.created_at);
      const today = new Date();
      const daysSinceLastOrder = Math.floor((today - lastOrderDate) / (1000 * 60 * 60 * 24));
 
      // Puntaje basado en días desde última orden
      if (daysSinceLastOrder <= 30) return 25; // Excelente
      if (daysSinceLastOrder <= 90) return 20; // Muy bueno
      if (daysSinceLastOrder <= 180) return 15; // Bueno
      if (daysSinceLastOrder <= 365) return 10; // Regular
      return 5; // Malo
    } catch (error) {
      console.error('Error calculando recency score:', error);
      return 0;
    }
  },
 
  // Calcular puntaje de frecuencia
  async calculateFrequencyScore(clientId) {
    try {
      const oneYearAgo = new Date();
      oneYearAgo.setFullYear(oneYearAgo.getFullYear() - 1);
 
      const { data: orders, error } = await supabase
        .from('ordenesdepublicidad')
        .select('id_ordenes_de_comprar')
        .eq('campania.clientes.id_cliente', clientId)
        .gte('created_at', oneYearAgo.toISOString());
 
      if (error) {
        console.warn('Error calculando frequency score, usando valor por defecto:', error);
        return 10; // Valor medio por defecto
      }
 
      const orderCount = orders?.length || 0;
 
      // Puntaje basado en número de órdenes en el último año
      if (orderCount >= 12) return 20; // Más de 1 orden por mes
      if (orderCount >= 6) return 15; // Más de 1 orden cada 2 meses
      if (orderCount >= 3) return 10; // Más de 1 orden cada 4 meses
      if (orderCount >= 1) return 5; // Al menos 1 orden
      return 0; // Sin órdenes
    } catch (error) {
      console.error('Error calculando frequency score:', error);
      return 0;
    }
  },
 
  // Calcular puntaje monetario
  async calculateMonetaryScore(clientId) {
    try {
      const { data: orders, error } = await supabase
        .from('ordenesdepublicidad')
        .select('campania!inner(Presupuesto)')
        .eq('campania.clientes.id_cliente', clientId);
 
      if (error) {
        console.warn('Error calculando monetary score, usando valor por defecto:', error);
        return 12; // Valor medio por defecto
      }
 
      const totalValue = orders?.reduce((sum, order) => {
        return sum + (order.campania?.Presupuesto || 0);
      }, 0) || 0;
 
      // Puntaje basado en valor total histórico
      if (totalValue >= 100000000) return 25; // Más de 100M
      if (totalValue >= 50000000) return 20; // Más de 50M
      if (totalValue >= 25000000) return 15; // Más de 25M
      if (totalValue >= 10000000) return 10; // Más de 10M
      if (totalValue >= 1000000) return 5; // Más de 1M
      return 0; // Menos de 1M
    } catch (error) {
      console.error('Error calculando monetary score:', error);
      return 0;
    }
  },
 
  // Calcular puntaje de éxito de campañas
  async calculateCampaignSuccessScore(clientId) {
    try {
      const { data: campaigns, error } = await supabase
        .from('campania')
        .select('estado')
        .eq('id_Cliente', clientId);
 
      if (error) {
        console.warn('Error calculando campaign success score, usando valor por defecto:', error);
        return 10; // Valor medio por defecto
      }
 
      if (!campaigns || campaigns.length === 0) return 0;
 
      const successfulCampaigns = campaigns.filter(c => c.estado === 'finalizada').length;
      const successRate = (successfulCampaigns / campaigns.length) * 100;
 
      // Puntaje basado en tasa de éxito de campañas
      if (successRate >= 90) return 15; // Excelente
      if (successRate >= 75) return 12; // Muy bueno
      if (successRate >= 60) return 9; // Bueno
      if (successRate >= 45) return 6; // Regular
      if (successRate >= 30) return 3; // Malo
      return 0; // Muy malo
    } catch (error) {
      console.error('Error calculando campaign success score:', error);
      return 0;
    }
  },
 
  // Calcular puntaje de retención
  async calculateRetentionScore(clientId) {
    try {
      const twoYearsAgo = new Date();
      twoYearsAgo.setFullYear(twoYearsAgo.getFullYear() - 2);
 
      const { data: orders, error } = await supabase
        .from('ordenesdepublicidad')
        .select('created_at')
        .eq('campania.clientes.id_cliente', clientId)
        .gte('created_at', twoYearsAgo.toISOString())
        .order('created_at', { ascending: true });
 
      if (error) {
        console.warn('Error calculando retention score, usando valor por defecto:', error);
        return 8; // Valor medio por defecto
      }
 
      if (!orders || orders.length < 2) return 0;
 
      // Calcular consistencia de pedidos
      const orderDates = orders.map(o => new Date(o.created_at));
      const intervals = [];
 
      for (let i = 1; i < orderDates.length; i++) {
        const interval = (orderDates[i] - orderDates[i-1]) / (1000 * 60 * 60 * 24);
        intervals.push(interval);
      }
 
      const avgInterval = intervals.reduce((sum, interval) => sum + interval, 0) / intervals.length;
      const consistency = Math.max(0, 365 - avgInterval) / 365; // 1 = muy consistente, 0 = inconsistente
 
      return Math.round(consistency * 15);
    } catch (error) {
      console.error('Error calculando retention score:', error);
      return 0;
    }
  },

  getClientLevel(score) {

      if (score >= level.minScore) {
        return level;
      }
    }
    return this.clientLevels.bronze;

  async saveClientScoring(scoringData) {

      // No lanzamos error para no interrumpir el flujo
    }
  },
 
  // Obtener scoring de todos los clientes
  async getAllClientScores() {
    try {
      const { data: clients, error } = await supabase
        .from('clientes')
        .select('id_cliente, nombrecliente, razonSocial');
 
      if (error) throw error;
 
      const clientScores = await Promise.all(
        clients.map(async (client) => {
          try {
            const score = await this.calculateClientScore(client.id_cliente);
            return {
              ...client,
              ...score
            };
          } catch (error) {
            console.error(`Error calculando score para cliente ${client.id_cliente}:`, error);
            return {
              ...client,
              totalScore: 0,
              level: 'Sin datos',
              error: true
            };
          }
        })
      );
 
      return clientScores.sort((a, b) => b.totalScore - a.totalScore);
    } catch (error) {
      console.error('Error obteniendo scores de clientes:', error);
      return [];
    }
  },
 
  // Obtener scoring de un cliente específico
  async getClientScore(clientId) {
    try {
      const { data: existingScore, error } = await supabase
        .from('client_scoring')
        .select('*')
        .eq('client_id', clientId)
        .single();
 
      if (error && error.code !== 'PGRST116') throw error;
 
      // Si existe y es reciente (menos de 24 horas), devolverlo
      if (existingScore) {
        const calculatedAt = new Date(existingScore.calculated_at);
        const now = new Date();
        const hoursDiff = (now - calculatedAt) / (1000 * 60 * 60);
 
        if (hoursDiff < 24) {
          return existingScore;
        }
      }
 
      // Calcular nuevo scoring
      return await this.calculateClientScore(clientId);
    } catch (error) {
      console.error('Error obteniendo score de cliente:', error);
      throw error;
    }
  },

  // Obtener tasa de retención de clientes
  async getClientRetentionRate() {
    try {
      const oneYearAgo = new Date();
      oneYearAgo.setFullYear(oneYearAgo.getFullYear() - 1);
 
      const [clientsLastYear, activeClientsThisYear] = await Promise.all([
        // Clientes que tenían órdenes el año pasado
        supabase
          .from('ordenesdepublicidad')
          .select('campana!inner(id_Cliente)')
          .gte('created_at', oneYearAgo.toISOString())
          .lt('created_at', new Date().toISOString()),
        
        // Clientes que tienen órdenes este año
        supabase
          .from('ordenesdepublicidad')
          .select('campana!inner(id_Cliente)')
          .gte('created_at', new Date(new Date().getFullYear(), 0, 1).toISOString())
      ]);
 
      if (!clientsLastYear.data || clientsLastYear.data.length === 0) return 100;
 
      // Extraer IDs únicos de clientes
      const clientIdsLastYear = new Set(
        clientsLastYear.data.map(order => order.campana?.id_Cliente).filter(Boolean)
      );
      
      const activeClientIdsThisYear = new Set(
        activeClientsThisYear.data?.map(order => order.campana?.id_Cliente).filter(Boolean) || []
      );
 
      // Calcular clientes retenidos (clientes del año pasado que siguen activos)
      let retainedClients = 0;
      clientIdsLastYear.forEach(clientId => {
        if (activeClientIdsThisYear.has(clientId)) {
          retainedClients++;
        }
      });
 
      const retentionRate = (retainedClients / clientIdsLastYear.size) * 100;
      return Math.round(retentionRate);
    } catch (error) {
      console.error('Error calculando tasa de retención:', error);
      return 0;
    }
  }
